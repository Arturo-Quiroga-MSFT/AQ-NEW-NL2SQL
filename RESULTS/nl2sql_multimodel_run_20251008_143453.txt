SELECT
    c.CollateralId,
    c.Description,
    c.ValueAmount,
    co.CompanyName
FROM
    dbo.Collateral c
    INNER JOIN dbo.Loan l ON c.LoanId = l.LoanId
    INNER JOIN dbo.Company co ON l.CompanyId = co.CompanyId
WHERE
    c.ValueAmount > 1000000
