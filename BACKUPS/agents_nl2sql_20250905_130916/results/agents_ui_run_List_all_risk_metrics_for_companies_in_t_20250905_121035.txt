Query: List all risk metrics for companies in the Technology industry.

Intent & Entities (JSON):
{
  "intent": "List all risk metrics for companies in the Technology industry.",
  "error": "'\"intent\"'"
}

Generated SQL:
WITH TechCompanies AS (
  SELECT c.CompanyId, c.CompanyName
  FROM dbo.Company c
  WHERE UPPER(LTRIM(RTRIM(ISNULL(c.Industry, '')))) = 'TECHNOLOGY'
)
SELECT r.CompanyId, tc.CompanyName, r.AsOfDate, r.MetricName, r.MetricValue, r.Source
FROM dbo.RiskMetricHistory r
JOIN TechCompanies tc ON r.CompanyId = tc.CompanyId
ORDER BY tc.CompanyName, r.AsOfDate DESC, r.MetricName
