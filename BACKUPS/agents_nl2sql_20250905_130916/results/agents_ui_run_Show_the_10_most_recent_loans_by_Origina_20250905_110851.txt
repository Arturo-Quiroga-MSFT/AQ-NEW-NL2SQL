Query: Show the 10 most recent loans by OriginationDate.

Generated SQL:
SELECT
  lp.LoanId,
  lp.LoanNumber,
  lp.CompanyName,
  lp.Industry,
  lp.CreditRating,
  lp.CountryName,
  lp.RegionName,
  lp.OriginationDate,
  lp.MaturityDate,
  lp.PrincipalAmount,
  lp.CurrencyCode,
  lp.InterestRatePct,
  lp.InterestRateType,
  lp.ReferenceRate,
  lp.SpreadBps,
  lp.AmortizationType,
  lp.PaymentFreqMonths,
  lp.Status,
  lp.Purpose,
  np.DueDate AS NextDueDate,
  np.TotalDue AS NextTotalDue,
  np.PrincipalDue AS NextPrincipalDue,
  np.InterestDue AS NextInterestDue,
  np.PaymentCurrency
FROM dbo.vw_LoanPortfolio AS lp
OUTER APPLY (
  SELECT TOP (1)
    ps.DueDate,
    ps.TotalDue,
    ps.PrincipalDue,
    ps.InterestDue,
    ps.CurrencyCode AS PaymentCurrency
  FROM dbo.PaymentSchedule AS ps
  WHERE ps.LoanId = lp.LoanId
    AND (ps.PaidFlag = 0 OR ps.PaidFlag IS NULL)
    AND (ps.Status IS NULL OR ps.Status NOT IN ('Cancelled','Skipped'))
  ORDER BY ps.DueDate ASC
) AS np
ORDER BY lp.OriginationDate DESC, lp.LoanId;
